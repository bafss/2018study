mount挂载外部存储设备到文件系统中
umount /mnt/cdrom
     mkdir /mnt/cdrom
     mount -t -iso9660 -o ro /dev/cdrom /mnt/cdrom
        -t  设备类型
        -o  读的类型 ro只读
